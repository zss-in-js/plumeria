## Overview


This includes APIs exposed as CSS-in-JS.

```ts title="TypeScript"
export { css, rx } from './css';
export type { CreateStyle, CSSProperties } from './css';
```

## Use API
```ts title="TypeScript"
import { css, rx } from 'plumeria/core';
import type { CSSProperties, CreateStyle } from 'plumeria/core';
```

## Types Reference

### `CSSProperties`

```ts
import type { CSSProperties } from '@plumeria/core';

const styles: CSSProperties = {
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'space-between',
  backgroundColor: 'var(--color-bg)',
  padding: '1rem',
  ':hover': {
    backgroundColor: 'var(--color-bg-hover)',
  },
  '@media (min-width: 768px)': {
    flexDirection: 'row',
  },
};
```

### `CreateStyle`
```ts
import type { CreateStyle } from '@plumeria/core';

const styles: CreateStyle = {
  container: {
    maxWidth: '1280px',
    margin: '0 auto',
  },
  title: {
    fontSize: '2rem',
    fontWeight: 'bold',
  },
};
```
