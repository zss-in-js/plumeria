[`@plumeria/turbopack-loader`](https://www.npmjs.com/package/@plumeria/turbopack-loader) is the Turbopack loader for Plumeria.  
It compiles styles into CSS via AST parsing, with HMR support.

## Installation
<Tabs items={['npm',  'yarn', 'pnpm']}>
<Tab>
```sh title="Terminal"
npm i --save-dev @plumeria/turbopack-loader
```
</Tab>
<Tab>
```sh title="Terminal"
yarn add -D @plumeria/turbopack-loader
```
</Tab>
<Tab>
```sh title="Terminal"
pnpm i --save-dev @plumeria/turbopack-loader
```
</Tab>
</Tabs>

### Configuration in Next.js project
If you use Plumeria with Next.js, customize the `turbopack` in `next.config.ts`:
```ts title="next.config.ts"
import type { NextConfig, TurbopackLoaderItem } from 'next/dist/server/config-shared';

const reactLoaders: TurbopackLoaderItem[] = [
  {
    loader: '@plumeria/turbopack-loader',
    options: {},
  },
];

const nextConfig: NextConfig = {
  turbopack: {
    rules: {
      '*.ts': {
        loaders: reactLoaders,
      },
      '*.tsx': {
        loaders: reactLoaders,
      },
      '*.js': {
        loaders: reactLoaders,
      },
      '*.jsx': {
        loaders: reactLoaders,
      },
    },
  },
};

export default nextConfig;
```
