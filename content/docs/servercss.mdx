## Headless component

A headless component for facilitating server-side js for Next.js only.

<Callout title="It is safe to know">
  It will be null in non-development environments and will not be included in
  production environments.
</Callout>

```tsx title="layout.tsx"
import { ServerCSS } from '@plumeria/next'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <head> // [!code ++]
        <ServerCSS /> // [!code ++]
      </head> // [!code ++]
      <body>
        <main>{children}</main>
      </body>
    <html>
  )
}
```
