### Compile command

<code>npx css</code> compiles without type checking. Adding the argument
<code>--type-check</code>
will include compilation checking using TypeScript. You can also add the
<code>--log</code>
argument to see the CSS log that is output during compilation. It is recommended
to generating server-side static CSS with a faster compile In parallel next dev.

<Tabs items={["Next.js"]}>
<Tab>
```json title="package.json"
  "scripts": {
    "dev": "css & next dev",
    "build": "css --type-check && next build",
  }
```
</Tab>
</Tabs>

### Server component

You can statically preview a Next.js's ServerComponent using <code>\<ServerCSS \/\></code>.  
This ServerCSS is a headless component and is not included in the production environment.

<Tabs items={["Next.js"]}>
<Tab>
```tsx title="layout.tsx"
import { ServerCSS } from '@plumeria/next'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <head>
        <ServerCSS /> // [!code ++]
      </head> 
      <body>
        <main>{children}</main>
      </body>
    </html>
  )
}
```
</Tab>
</Tabs>

## Linting

Uses the community-made [eslint-plugin-object-css](https://www.npmjs.com/package/eslint-plugin-object-css).

<Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>
<Tab>
```sh title="shell"
npm install --save-dev eslint-plugin-object-css
```
</Tab>
<Tab>
```sh title="shell"
pnpm install --save-dev eslint-plugin-object-css
```
</Tab>
<Tab>
```sh title="shell"
yarn add -D eslint-plugin-object-css
```
</Tab>
<Tab>
```sh title="shell"
bun add -D eslint-plugin-object-css
```
</Tab>
</Tabs>
<Tabs items={['eslint8', 'eslint9']}>
<Tab>
```json title=".eslintrc.json"
{
  "extends": ["plugin:object-css/recommended"],  
  "plugins": ["object-css"],
  "rules": {}
}
```
</Tab>
  <Tab>
    ```js title="eslint.config.mjs"
    import { FlatCompat } from '@eslint/eslintrc'
    import objectCss from 'eslint-plugin-object-css'

    const __dirname = import.meta.dirname;
    const compat = new FlatCompat({
      baseDirectory: __dirname,
    })

    const eslintConfig = [
      {
        files: ['**/*.{ts,js,jsx,tsx}'],
        plugins: {
          'object-css': objectCss,
        },
        rules: {
          'object-css/recess-order': 'warn',
          'object-css/valid-value': 'error',
        },
      },
    ]

    export default eslintConfig
    ```

  </Tab>
</Tabs>
